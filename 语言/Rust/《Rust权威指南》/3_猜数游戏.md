
```Rust
use rand::Rng;//Rng是一个特征（trait）；可以利用cargo doc --open阅读库中的其他函数

use std::cmp::Ordering; //Ordering是一个枚举类型，有三个变体，用于表示比较数字之后产生的结果

use std::io; //用于导入标准库的io模块 

//添加外部包的流程：在cargo.toml文件中的[dependencies]下添加库包名和版次；Cargo.lock文件会记录依赖和版次，安装对应依赖，确保项目的可复现性

//利用cargo update进行更新，且仅能更新一个版本

fn main() {

  

let secret_number = rand::thread_rng().gen_range(1, 101); //rand::thread_rng()会返回一个特定的随机数生成器，通过操作系统返回一个随机数

  

loop {

println!("Guess the number!");

  

println!("Please input your guess.");

  

let mut guess = String::new(); //创建出一个可变的变量，并绑定了一个空白字符串在它身上

//new是String类型的一个关联函数；在Rust中，针对于类型本身来定义关联函数，叫做静态方法

io::stdin()

.read_line(&mut guess) //stdin（）标准输入函数；read_line（&String）接受输入并传参带目标字符串，需要引用

.expect("Failed to read line"); //readline()方法还会返回一个io::result的值；

//Result类型有两种返回结果：OK和Err，Err时打印expect()方法的字符串

  

let guess: u32 = match guess.trim().parse() {//trim()方法用于去除多余的空白字符，parse()用于将字符串转换为数字

Ok(num) => num,

Err(_) =>continue,

};

  

println!("Your guess: {}", guess);

  

match guess.cmp(&secret_number) {

//match表达式由数个分支组成，每个分支都包含一个用于匹配的分支

Ordering::Less => println!("Too Small!"),

Ordering::Greater => println!("Too big!"),

Ordering::Equal => {

println!("You win!");

break;

}

}

}

}
```
